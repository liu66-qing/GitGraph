# PulseGraph 项目升级方案

> 从 CodeGraph（通用知识图谱演化系统）升级为 PulseGraph（多源信息脉络图谱智能分析系统）
> 目标：产品场景落地 + 底层技术深度达到 Agent 研发工程师面试竞争力

---

## 一、产品定位与命名

### 1.1 命名

- 产品名：**PulseGraph（脉图）**
- 一句话定位：从碎片化多源信息中自动构建事件脉络与人物关系图谱的 AI 分析系统
- "脉"的双关：脉络（事件线索）+ 脉搏（实时更新）

### 1.2 场景覆盖（不绑定单一领域）

| 场景 | 示例 | 展示优先级 |
|------|------|-----------|
| 国际新闻/地缘政治 | 美伊冲突各方关系与事件升级链 | P0（首个 demo） |
| 明星八卦/娱乐事件 | 人物关系、前因后果、各方回应 | P1（后续扩展） |
| 影视剧/小说 | 角色关系图、剧情因果链 | P2 |
| 商业/科技新闻 | 公司并购、人事变动、竞争格局 | P2 |

### 1.3 核心用户故事

> 用户看到一条抖音新闻视频或一篇报道，想搞清楚"这些人什么关系""事情怎么发展到这一步的"。
> 把链接丢进来，系统自动：
> 1. 提取视频/文章内容
> 2. 主动搜索补全相关报道
> 3. 构建人物关系图 + 事件时间线
> 4. 标注信息冲突和来源可靠度
> 5. 支持自然语言追问

```python
"""URL 内容抓取器：新闻链接 → 正文文本"""

import httpx
from bs4 import BeautifulSoup

class URLLoader:
    async def load(self, url: str) -> dict:
        # 1. 请求网页
        async with httpx.AsyncClient() as client:
            resp = await client.get(url, follow_redirects=True, timeout=15)
        
        # 2. 提取正文（去除导航、广告等）
        soup = BeautifulSoup(resp.text, "html.parser")
        for tag in soup(["script", "style", "nav", "footer", "header", "aside"]):
            tag.decompose()
        
        # 3. 提取元数据
        title = soup.find("title").get_text(strip=True) if soup.find("title") else ""
        text = soup.get_text(separator="\n", strip=True)
        
        return {
            "text": f"标题：{title}\n\n{text}",
            "filename": f"url_{hash(url)}.txt",
            "metadata": {
                "source_type": "web_article",
                "source_url": url,
                "reliability_score": 0.8,  # 默认值，后续按域名调整
            }
        }
```

### 2.3 抖音视频内容提取

技术方案：标题文案 + ASR 语音转写

```python
"""抖音视频内容提取器：链接 → 文案 + ASR 转写文本"""

import subprocess
import tempfile
from pathlib import Path
import structlog

logger = structlog.get_logger()


class DouyinLoader:
    """
    流程：解析链接 → 获取元数据（标题+文案+作者）→ 下载视频 → 提取音频 → ASR → 合并
    """

    async def extract(self, douyin_url: str) -> dict:
        # 1. 获取视频元数据
        metadata = await self._get_metadata(douyin_url)

        # 2. 下载视频 + 提取音频
        audio_path = await self._download_and_extract_audio(douyin_url)

        # 3. ASR 语音转文字
        transcript = ""
        if audio_path:
            transcript = await self._transcribe(audio_path)
            audio_path.unlink(missing_ok=True)  # 清理临时文件

        # 4. 合并所有文本
        full_text = self._merge_text(metadata, transcript)

        return {
            "text": full_text,
            "filename": f"douyin_{metadata.get('video_id', 'unknown')}.txt",
            "metadata": {
                "source_type": "douyin",
                "source_url": douyin_url,
                "author": metadata.get("author", ""),
                "publish_time": metadata.get("publish_time", ""),
                "reliability_score": self._calc_reliability(metadata),
            }
        }

    async def _get_metadata(self, url: str) -> dict:
        """用 yt-dlp 获取视频元数据（无需下载视频）"""
        cmd = ["yt-dlp", "--dump-json", "--no-download", url]
        result = subprocess.run(cmd, capture_output=True, text=True, timeout=30)
        if result.returncode == 0:
            import json
            data = json.loads(result.stdout)
            return {
                "video_id": data.get("id", ""),
                "title": data.get("title", ""),
                "description": data.get("description", ""),
                "author": data.get("uploader", ""),
                "publish_time": data.get("upload_date", ""),
                "duration": data.get("duration", 0),
                "like_count": data.get("like_count", 0),
                "is_verified": False,  # 需要额外判断
            }
        return {}

    async def _download_and_extract_audio(self, url: str) -> Path | None:
        """下载视频并提取音频轨道"""
        tmp_dir = tempfile.mkdtemp()
        audio_path = Path(tmp_dir) / "audio.wav"
        
        # yt-dlp 下载 + ffmpeg 提取音频（一步完成）
        cmd = [
            "yt-dlp", "-x",  # 只提取音频
            "--audio-format", "wav",
            "--audio-quality", "0",
            "-o", str(audio_path.with_suffix("")),  # yt-dlp 会自动加后缀
            url
        ]
        result = subprocess.run(cmd, capture_output=True, timeout=120)
        
        # yt-dlp 可能输出为 .wav 或其他格式，找到实际文件
        for f in Path(tmp_dir).glob("audio.*"):
            return f
        return None

        # 方案B：DashScope ASR API（阿里云，中文效果好，项目已用 DashScope）
        # 方案C：火山引擎 ASR API
        pass

    def _merge_text(self, metadata: dict, transcript: str) -> str:
        """合并元数据和转写文本"""
        parts = []
        if metadata.get("title"):
            parts.append(f"【标题】{metadata['title']}")
        if metadata.get("description"):
            parts.append(f"【文案】{metadata['description']}")
        if transcript:
            parts.append(f"【视频口播内容】\n{transcript}")
        parts.append(f"【来源】抖音 @{metadata.get('author', '未知')}")
        parts.append(f"【发布时间】{metadata.get('publish_time', '未知')}")
        return "\n\n".join(parts)

    def _calc_reliability(self, metadata: dict) -> float:
        """根据账号属性计算来源可靠度"""
        if metadata.get("is_verified"):
            return 0.7
        if metadata.get("like_count", 0) > 100000:
            return 0.6
        return 0.4
```

### 3.4 相关性判断

```python
class RelevanceScorer:
    """判断候选内容是否与目标事件相关"""
    
    RELEVANCE_THRESHOLD = 0.5  # 低于此分数的内容不纳入
    
    def score(self, candidate_text: str, seed_context: dict) -> float:
        """
        评分维度（加权求和）：
        - 实体重叠度 0.4：候选内容中出现了多少个种子实体
        - 事件关键词匹配 0.3：是否提到核心事件
        - 时间接近度 0.2：发布时间是否在事件时间窗口内
        - 语义相似度 0.1：embedding 余弦相似度
        """
        entity_overlap = self._entity_overlap(candidate_text, seed_context["entities"])
        keyword_match = self._keyword_match(candidate_text, seed_context["keywords"])
        time_proximity = self._time_proximity(candidate_text, seed_context["time_range"])
        semantic_sim = self._semantic_similarity(candidate_text, seed_context["embedding"])
        
        return 0.4 * entity_overlap + 0.3 * keyword_match + 0.2 * time_proximity + 0.1 * semantic_sim
```

### 3.5 饱和度判断（停止条件）

```python
class SaturationDetector:
    """判断信息是否饱和，决定是否停止搜索"""
    
    MAX_CONTENT = 15          # 硬上限：最多处理15条内容
    MAX_ROUNDS = 4            # 最多搜索4轮
    STALE_THRESHOLD = 2       # 连续2轮无新信息则停止
    TIMEOUT_SECONDS = 180     # 总超时3分钟
    
    def __init__(self):
        self.known_entities: set = set()
        self.known_relations: set = set()
        self.rounds_without_gain = 0
        self.total_processed = 0
    
    def should_stop(self, new_extraction) -> tuple[bool, str]:
        """返回 (是否停止, 停止原因)"""
        
        # 计算信息增量
        new_entities = set(e.name for e in new_extraction.entities) - self.known_entities
        new_relations = set(
            f"{r.source}-{r.type}-{r.target}" for r in new_extraction.relations
        ) - self.known_relations
        
        self.known_entities.update(new_entities)
        self.known_relations.update(new_relations)
        self.total_processed += 1
        
        info_gain = len(new_entities) + len(new_relations)
        
        if info_gain == 0:
            self.rounds_without_gain += 1
        else:
            self.rounds_without_gain = 0
        
        # 停止条件判断
        if self.rounds_without_gain >= self.STALE_THRESHOLD:
            return True, "信息饱和：连续2轮无新发现"
        if self.total_processed >= self.MAX_CONTENT:
            return True, f"达到内容上限：{self.MAX_CONTENT}条"
        if self._core_dimensions_covered():
            return True, "核心维度已覆盖"
        
        return False, ""
    
    def _core_dimensions_covered(self) -> bool:
        """检查 who/what/when/why/stance 五个维度是否都有信息"""
        has_people = len([e for e in self.known_entities]) >= 3
        has_relations = len(self.known_relations) >= 5
        return has_people and has_relations
```

## 五、底层技术深化（Agent 研发岗竞争力）

### 5.1 Prompt Engineering 升级

**现状问题**：prompt 是零-shot，无示例，输出不稳定时直接失败。

**升级要点**：

1. **Few-shot 示例**：每个 prompt 加 1~2 个领域示例（中文新闻抽取示例）
2. **输出格式约束**：明确 JSON Schema，告诉 LLM "只输出 JSON，不要其他内容"
3. **鲁棒解析器**：LLM 输出不合法时自动修复（提取代码块、找首尾大括号、LLM 修复）
4. **Prompt 版本管理**：prompt 从代码中抽出，放到 `src/codegraph/prompts/` 目录，YAML 格式，支持 A/B 对比
5. **动态 prompt 组装**：根据场景（国际新闻/娱乐/影视）加载不同的关系类型列表和 few-shot 示例

文件结构：

### 5.3 Agent 编排升级：显式状态机 + 结构化工作记忆

**现状问题**：简单 for 循环，working_memory 是无结构的 list，状态不可追踪。

**升级方案**：

```python
# ===== 状态机 =====
class AgentState(str, Enum):
    PLANNING = "planning"
    EXECUTING = "executing"
    SYNTHESIZING = "synthesizing"
    VALIDATING = "validating"
    REPLANNING = "replanning"
    SEARCHING = "searching"       # 主动搜索更多信息
    DONE = "done"
    FAILED = "failed"


# ===== 结构化工作记忆 =====
@dataclass
class WorkingMemory:
    question: str
    intent: str = ""
    
    # 按类型分类的证据
    graph_facts: list[dict] = field(default_factory=list)
    text_chunks: list[dict] = field(default_factory=list)
    temporal_facts: list[dict] = field(default_factory=list)
    conflicts: list[dict] = field(default_factory=list)
    
    # 执行追踪
    steps_executed: list[dict] = field(default_factory=list)
    state_history: list[str] = field(default_factory=list)
    
    # 资源消耗
    total_tokens_used: int = 0
    total_tool_calls: int = 0
    iteration_count: int = 0


# ===== 执行引擎 =====
class AgentExecutionEngine:
    def __init__(self, max_iterations: int = 5, token_budget: int = 50000):
        self.max_iterations = max_iterations
        self.token_budget = token_budget
    
    async def run(self, question: str) -> AgentResponse:
        memory = WorkingMemory(question=question)
        state = AgentState.PLANNING
        
        while state not in (AgentState.DONE, AgentState.FAILED):
            memory.state_history.append(state.value)
            
            match state:
                case AgentState.PLANNING:
                    state = await self._plan(memory)
                case AgentState.EXECUTING:
                    state = await self._execute(memory)
                case AgentState.SYNTHESIZING:
                    state = await self._synthesize(memory)
                case AgentState.VALIDATING:
                    state = await self._validate(memory)
                case AgentState.REPLANNING:
                    state = await self._replan(memory) if memory.iteration_count < self.max_iterations else AgentState.DONE
                case AgentState.SEARCHING:
                    state = await self._search_more(memory)
            
            # 安全阀
            if memory.total_tokens_used >= self.token_budget:
                state = AgentState.DONE
            memory.iteration_count += 1
        
        return self._build_response(memory)
```
